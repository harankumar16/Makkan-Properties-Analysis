def treat_outliers(data):
    q1 = np.percentile(data,25)
    q3 = np.percentile(data,75)
    iqr = q3-q1
    upper = q3 + (iqr*1.5)
    lower = q1 - (iqr*1.5)
    for i in range(0,len(data)):
        if (data[i]>upper) or (data[i]<lower):
            if (data[i]>upper):
                data[i] = upper
            elif (data[i]<lower):
                data[i]=lower
    return data
